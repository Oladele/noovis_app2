Create a network site
---------------------

Parameters:
  * type - Should always be set to <code>network-sites</code>
  * company - Company id
  * name - Network Site Name
  * lat - Network Site Latitude
  * lng - Network Site Longitude
  * address - Address Field

Request:
  POST /network-sites
  Content-Type: application/vnd.api+json

  {"data":{"type":"network-sites","relationships":{"company":{"data":{"type":"companies","id":462}}},"attributes":{"name":"Site Name"}}}

Response:
  Status: 201 Created
  Content-Type: application/vnd.api+json

  {"data":{"id":"291","type":"network-sites","links":{"self":"http://example.org/network-sites/291"},"attributes":{"name":"Site Name","lat":null,"lng":null,"address":null,"node-counts":[{"node_type":"buildings","count":0,"node_type_pretty":"Buildings"}]},"relationships":{"company":{"links":{"self":"http://example.org/network-sites/291/relationships/company","related":"http://example.org/network-sites/291/company"}},"buildings":{"links":{"self":"http://example.org/network-sites/291/relationships/buildings","related":"http://example.org/network-sites/291/buildings"}},"workbooks":{"links":{"self":"http://example.org/network-sites/291/relationships/workbooks","related":"http://example.org/network-sites/291/workbooks"}}}}}
