Update a sheet
--------------

Parameters:
  * workbook - Workbook id
  * building - Building id
  * type - Should always be set to <code>sheets</code>
  * name - Sheet Name
  * created-at - Sheet created at
  * updated-at - Sheet created at
  * record-count - Cable Run count for Sheet
  * id - The id of the sheet

Request:
  PATCH /sheets/286
  Content-Type: application/vnd.api+json

  {"data":{"relationships":{"workbook":{"data":{"type":"workbooks","id":364}},"building":{"data":{"type":"buildings","id":368}}},"type":"sheets","attributes":{"name":"Updated Sheet Name"},"id":"286"}}

Response:
  Status: 200 OK
  Content-Type: application/vnd.api+json

  {"data":{"id":"286","type":"sheets","links":{"self":"http://example.org/sheets/286"},"attributes":{"name":"Updated Sheet Name","created-at":"2016-04-09T17:35:52.593Z","updated-at":"2016-04-09T17:35:52.618Z","record-count":0},"relationships":{"workbook":{"links":{"self":"http://example.org/sheets/286/relationships/workbook","related":"http://example.org/sheets/286/workbook"}},"building":{"links":{"self":"http://example.org/sheets/286/relationships/building","related":"http://example.org/sheets/286/building"}},"cable-runs":{"links":{"self":"http://example.org/sheets/286/relationships/cable-runs","related":"http://example.org/sheets/286/cable-runs"}},"network-graphs":{"links":{"self":"http://example.org/sheets/286/relationships/network-graphs","related":"http://example.org/sheets/286/network-graphs"}}}}}
