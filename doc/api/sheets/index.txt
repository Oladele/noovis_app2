Update a sheet
--------------

Parameters:
  * workbook - Workbook id
  * building - Building id
  * type - Should always be set to <code>sheets</code>
  * name - Sheet Name
  * created-at - Sheet created at
  * updated-at - Sheet created at
  * record-count - Cable Run count for Sheet
  * id - The id of the sheet

Request:
  PATCH /sheets/355
  Content-Type: application/vnd.api+json

  {"data":{"relationships":{"workbook":{"data":{"type":"workbooks","id":452}},"building":{"data":{"type":"buildings","id":462}}},"type":"sheets","attributes":{"name":"Updated Sheet Name"},"id":"355"}}

Response:
  Status: 200 OK
  Content-Type: application/vnd.api+json

  {"data":{"id":"355","type":"sheets","links":{"self":"http://example.org/sheets/355"},"attributes":{"name":"Updated Sheet Name","created-at":"2016-04-09T18:59:06.947Z","updated-at":"2016-04-09T18:59:06.967Z","record-count":0},"relationships":{"workbook":{"links":{"self":"http://example.org/sheets/355/relationships/workbook","related":"http://example.org/sheets/355/workbook"}},"building":{"links":{"self":"http://example.org/sheets/355/relationships/building","related":"http://example.org/sheets/355/building"}},"cable-runs":{"links":{"self":"http://example.org/sheets/355/relationships/cable-runs","related":"http://example.org/sheets/355/cable-runs"}},"network-graphs":{"links":{"self":"http://example.org/sheets/355/relationships/network-graphs","related":"http://example.org/sheets/355/network-graphs"}}}}}
